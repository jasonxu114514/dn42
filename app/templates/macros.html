{# Reusable presentation macros shared by every page. - badge / status_badge / deploy_badge / bool_pill: coloured pills for states - stat_card: dashboard metric card - page_header: title + optional subtitle - code_block: 
 with a copy button (multi-line configs)
  - copy_button: standalone copy button (short values like tokens/keys)
  - pagination: prev / page-of / next pager
#}

{% macro badge(text, tone="neutral") %}{{ text }}{% endmacro %}

{% macro status_badge(status) %}
  {%- set tones = {"approved": "ok", "disabled": "bad", "pending": "warn"} -%}
  {{ badge(status, tones.get(status, "neutral")) }}
{%- endmacro %}

{% macro deploy_badge(status) %}
  {%- set tones = {"deployed": "ok", "failed": "bad", "deploying": "warn"} -%}
  {{ badge(status.replace("_", " "), tones.get(status, "neutral")) }}
{%- endmacro %}

{% macro bool_pill(value, on="yes", off="no") %}
  {{ badge(on, "ok") if value else badge(off, "neutral") }}
{%- endmacro %}
